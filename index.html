OSNOVNE KOMANDE TERMINALA

cd ime_direktorijuma ******** prelazak u izabrani direktorijum

cd .. ******** povratak u polazni(parent) direktorijum

cd dir ******** izlistavanje sadržaja direktorijuma

cd ls ******** izlistavanje sadržaja direktorijuma

clear ili cls ******** brisanje sadržaja ekrana

mkdir ime_foldera ili md ime_foldera ******** pravljenje foldera u trenutnom folderu

rmdir ime_foldera ili rd ime_foldera ******** brisanje foldera u trenutnom folderu

touch ime_fajla.tip ili ni ime_fajla.tip ******** kreiranje praznog fajla u trenutnom folderu

write "neki_kod" > ime_fajla ******** pisanje koda u fajl direktno iz komandne linije

KOMANDE ZA GIT I GITHUB

git config ************* konfiguriše git

git config --global user.name "Ime korisnika" ************* konfiguriše user name

git config --global user.email "user@example.com" ************* konfiguriše user email

git config naziv_podatka ************* proverava pojedinačni podatak

git config --list ************* izlistava podešavanja

git help naziv_komande ************* Pomoć u vezi komande

git init **** Od običnog projektnog foldera pravimo git projektni folder

git status **** Prikazivanje trenutnog statusa Git projekta

git add ime_fajla **** Git nepraćeni fajl prebacujemo u praćeni fajl

git add . **** Sve fajlove iz trenutnog foldera prebacuje u praćene fajlove

git commit -m"opis promene" **** Sve Čuvanje promena (commit)-trenutni snimak projekta

git log **** Prikaz (commit) promena

git log -1 **** Prikaz poslednje (commit) promene

git log -p **** Prikaz promena nastalih pri samim (commit) promenama

git log -- pretty = online **** Prikaz (commit) promena u jednoj liniji
